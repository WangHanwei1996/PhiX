$$
F=\int_V \left( f_{\mathrm{chem}}(c,\eta)+\frac{\kappa_c}{2}|\nabla c|^2+\frac{\kappa_\eta}{2}|\nabla \eta|^2 \right)\, dV
$$

$$
f_{\mathrm{chem}}(c,\eta)=f^{\alpha}(c)\left[1-h(\eta)\right]+f^{\beta}(c)\,h(\eta)+w\,g(\eta)
$$

$$
f^{\alpha}(c)=\varrho^{2}(c-c_{\alpha})^{2}
$$

$$
f^{\beta}(c)=\varrho^{2}(c_{\beta}-c)^{2}
$$

$$
h(\eta)=\eta^{3}(6\eta^{2}-15\eta+10)
$$

$$
g(\eta)=\eta^{2}(1-\eta)^{2}
$$
